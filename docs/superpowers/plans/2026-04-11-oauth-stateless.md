# OAuth Stateless Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Eliminate OAuth state storage from Express sessions, enabling the main session cookie to be upgraded from `sameSite: 'lax'` to `sameSite: 'strict'`, hardening CSRF protection.

**Architecture:** Replace Passport's default session-based state store with a custom HMAC-signed cookie store (`lib/oauth-state-store.js`). The OAuth state is signed with `SESSION_SECRET` and stored in a short-lived `oauth_state` cookie (sameSite: lax — required for the Google callback cross-site redirect). The main `rdi.sid` session cookie becomes `sameSite: strict`. No new npm dependencies.

**Tech Stack:** Express.js, Passport, passport-google-oauth20, Node.js `crypto` module

---

## Context

**Why sameSite: 'lax' now, and why it's a problem:**
The current setup stores Google OAuth state in the Express session. When Google redirects back to `/api/auth/google/callback`, the browser performs a cross-site GET. With `sameSite: 'lax'`, the session cookie IS sent (lax allows cross-site top-level GETs), so Passport can find the stored state. Changing to `'strict'` would prevent the cookie from being sent on that redirect, breaking the OAuth flow.

**The fix — two separate cookies:**
- `oauth_state` cookie: `sameSite: 'lax'`, lives 10 minutes, holds the HMAC-signed state. Sent on the Google callback. Cleared immediately after use.
- `rdi.sid` cookie: `sameSite: 'strict'`, lives 30 days, holds the session. Never sent cross-site.

**Security result:** CSRF is now impossible against all API endpoints. The `oauth_state` cookie itself is CSRF-safe because it's not a credential — it's a short-lived nonce, and its integrity is verified by HMAC.

**No new deps:** We avoid `cookie-parser` by using raw `req.headers.cookie` parsing and `res.setHeader('Set-Cookie', ...)` directly.

```js
// lib/oauth-state-store.js
'use strict';

const crypto = require('crypto');

const COOKIE_NAME  = 'oauth_state';
const MAX_AGE_SECS = 10 * 60; // 10 minutes

// ── Helpers ───────────────────────────────────────────────────────────────────

function hmac(value, secret) {
  return crypto.createHmac('sha256', secret).update(value).digest('base64url');
}

/**
 * Parse the raw Cookie header into a key→value map.
 * Avoids adding cookie-parser as a dependency.
 */
function parseCookies(header) {
  const map = {};
  if (!header) return map;
  for (const part of header.split(';')) {
    const idx = part.indexOf('=');
    if (idx < 0) continue;
    const key = part.slice(0, idx).trim();
    const val = part.slice(idx + 1).trim();
    map[key] = decodeURIComponent(val);
  }
  return map;
}

// ── CookieStateStore ──────────────────────────────────────────────────────────

/**
 * Custom Passport OAuth2 state store that keeps state in a short-lived
 * HMAC-signed cookie instead of the Express session.
 *
 * This allows the main session cookie to use sameSite: 'strict' while
 * the oauth_state cookie uses sameSite: 'lax' (required so Google can
 * send it back on the OAuth callback cross-site redirect).
 *
 * Interface matches passport-oauth2's AbstractStateStore:
 *   store(req, state, callback)
 *   verify(req, providedState, callback)
 */
class CookieStateStore {
  /**
   * Called by Passport when initiating the OAuth flow.
   * Signs `state` with HMAC and sets it as a short-lived cookie.
   *
   * @param {import('express').Request} req
   * @param {string} state - random nonce generated by passport-oauth2
   * @param {Function} callback - (err) => void
   */
  store(req, state, callback) {
    const secret = process.env.SESSION_SECRET;
    if (!secret) {
      return callback(new Error('SESSION_SECRET not set — cannot sign OAuth state'));
    }

    const sig    = hmac(state, secret);
    const value  = encodeURIComponent(`${state}.${sig}`);
    const secure = process.env.NODE_ENV === 'production' ? '; Secure' : '';

    req.res.setHeader('Set-Cookie',
      `${COOKIE_NAME}=${value}; HttpOnly; SameSite=Lax; Max-Age=${MAX_AGE_SECS}; Path=/${secure}`
    );

    callback(null);
  }

  /**
   * Called by Passport on the OAuth callback.
   * Reads the signed cookie, verifies HMAC + equality, clears the cookie.
   *
   * @param {import('express').Request} req
   * @param {string} providedState - state returned by Google in query string
   * @param {Function} callback - (err, state | false) => void
   */
  verify(req, providedState, callback) {
    const secret = process.env.SESSION_SECRET;
    if (!secret) {
      return callback(new Error('SESSION_SECRET not set'));
    }

    const cookies  = parseCookies(req.headers.cookie);
    const cookieVal = cookies[COOKIE_NAME];

    // Always clear the cookie, whether verification succeeds or fails
    req.res.setHeader('Set-Cookie',
      `${COOKIE_NAME}=; HttpOnly; SameSite=Lax; Max-Age=0; Path=/`
    );

    if (!cookieVal) {
      return callback(null, false, { message: 'OAuth state cookie missing — possible CSRF or expired flow' });
    }

    const dotIdx = cookieVal.lastIndexOf('.');
    if (dotIdx < 0) {
      return callback(null, false, { message: 'OAuth state cookie malformed' });
    }

    const state = cookieVal.slice(0, dotIdx);
    const sig   = cookieVal.slice(dotIdx + 1);

    // Constant-time comparison to prevent timing attacks
    const expectedSig = hmac(state, secret);
    let sigMatch;
    try {
      sigMatch = crypto.timingSafeEqual(
        Buffer.from(sig,         'base64url'),
        Buffer.from(expectedSig, 'base64url'),
      );
    } catch {
      sigMatch = false;
    }

    if (!sigMatch) {
      return callback(null, false, { message: 'OAuth state signature invalid' });
    }

    if (state !== providedState) {
      return callback(null, false, { message: 'OAuth state mismatch' });
    }

    callback(null, state);
  }
}

module.exports = { CookieStateStore };
```

node -e "
const { CookieStateStore } = require('./lib/oauth-state-store');
process.env.SESSION_SECRET = 'test-secret-min-32-chars-xxxxxxxxx';

const store = new CookieStateStore();
const capturedCookies = [];

// Mock req/res
const mockReq = (cookieHeader) => ({
  headers: { cookie: cookieHeader },
  res: {
    setHeader: (name, val) => { if (name === 'Set-Cookie') capturedCookies.push(val); }
  }
});

// Test 1: store sets a cookie
let req1 = mockReq('');
store.store(req1, 'my-state-nonce', (err) => {
  if (err) { console.error('FAIL store:', err.message); process.exit(1); }
  const cookie = capturedCookies[0];
  if (!cookie || !cookie.startsWith('oauth_state=')) {
    console.error('FAIL: cookie not set'); process.exit(1);
  }
  console.log('PASS: store sets cookie');

  // Extract the value to simulate what the browser would send back
  const val = decodeURIComponent(cookie.split('=')[1].split(';')[0]);

  // Test 2: verify accepts correct state
  let req2 = mockReq('oauth_state=' + encodeURIComponent(val));
  store.verify(req2, 'my-state-nonce', (err2, state, info) => {
    if (err2 || state !== 'my-state-nonce') {
      console.error('FAIL verify correct:', err2?.message, state, info); process.exit(1);
    }
    console.log('PASS: verify accepts correct state');

    // Test 3: verify rejects tampered state
    let req3 = mockReq('oauth_state=' + encodeURIComponent(val + 'tampered'));
    store.verify(req3, 'my-state-nonce', (err3, state3) => {
      if (state3 !== false) {
        console.error('FAIL verify tampered: expected false, got', state3); process.exit(1);
      }
      console.log('PASS: verify rejects tampered state');

      // Test 4: verify rejects missing cookie
      let req4 = mockReq('');
      store.verify(req4, 'my-state-nonce', (err4, state4) => {
        if (state4 !== false) {
          console.error('FAIL verify missing: expected false, got', state4); process.exit(1);
        }
        console.log('PASS: verify rejects missing cookie');
        console.log('\nAll 4 tests passed ✅');
      });
    });
  });
});
